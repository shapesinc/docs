import { useState } from "react";

<EngineFilter />

export const EngineFilter = () => {
  const engines = [
    // Premium Engines
    { name: "Anthropic: Claude Sonnet 4.5", tags: ["🧠","🎭","📚","🛠️"], description: "Favorite amongst roleplayers, 4.5 is intelligent, great for tool use, and Shape's default premium engine.", premium: true },
    { name: "DeepSeek V3.2 Exp", tags: ["🧠","📚","🛠️"], description: "The latest experimental model from DeepSeek.", premium: true },
    { name: "XAI: Grok 4 Fast", tags: ["🎭","📚"], description: "The faster, less expensive version of Grok 4.", premium: true },
    { name: "Gemini 2.5 Flash Image Preview", tags: ["🖼️"], description: "Image + text multimodal.", premium: true },
    { name: "OpenAI: GPT-5 Chat", tags: ["🎭","👤"], description: "ChatGPT for Roleplayers and chatters.", premium: true },

    // Free Engines
    { name: "OpenAI: GPT-5 Mini", tags: ["🎭","📚","🛠️"], description: "Balance of cost & capability.", premium: false },
    { name: "OpenAI: GPT-4.1 Mini", tags: ["🎭","📚"], description: "Smaller, cheaper, faster version of GPT-4.1.", premium: false },
    { name: "Grok 3 Mini", tags: ["🧠","🎭","📚","🛠️","👤"], description: "Amazing at following prompts, very fast, Swiss army knife of engines.", premium: false },
    { name: "OpenAI: GPT-4o-mini", tags: ["🎭","📚","👤"], description: "Fast GPT option, fallback for offline models.", premium: false }
  ];

  const allTags = [...new Set(engines.flatMap(e => e.tags))];
  const [selectedTag, setSelectedTag] = useState("");

  const filtered = selectedTag === ""
    ? engines
    : engines.filter(engine => engine.tags.includes(selectedTag));

  const premiumEngines = filtered.filter(e => e.premium);
  const freeEngines = filtered.filter(e => !e.premium);

  return (
    <div>
      <div style={{ marginBottom: "15px" }}>
        <select
          value={selectedTag}
          onChange={e => setSelectedTag(e.target.value)}
          style={{
            padding: "5px 12px",
            border: "1px solid #007bff",
            borderRadius: "5px",
            cursor: "pointer",
          }}
        >
          <option value="">All</option>
          {allTags.map(tag => (
            <option key={tag} value={tag}>{tag}</option>
          ))}
        </select>
      </div>

      {premiumEngines.length > 0 && (
        <div>
          <h2>Premium Engines</h2>
          <ul>
            {premiumEngines.map(engine => (
              <li key={engine.name} style={{ marginBottom: "10px", padding: "10px", borderRadius: "5px", transition: "background 0.2s" }}>
                <strong>{engine.name}</strong> [{engine.tags.join(", ")}]<br />
                {engine.description}
              </li>
            ))}
          </ul>
        </div>
      )}

      {freeEngines.length > 0 && (
        <div>
          <h2>Free Engines</h2>
          <ul>
            {freeEngines.map(engine => (
              <li key={engine.name} style={{ marginBottom: "10px", padding: "10px", borderRadius: "5px", transition: "background 0.2s" }}>
                <strong>{engine.name}</strong> [{engine.tags.join(", ")}]<br />
                {engine.description}
              </li>
            ))}
          </ul>
        </div>
      )}
    </div>
  );
};
