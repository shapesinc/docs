export function FilterButtons({ activeTags, setActiveTags }) {
  const tags = [
    { emoji: '🎭', label: 'Roleplay' },
    { emoji: '👤', label: 'Human-Like' },
    { emoji: '📚', label: 'Intelligent' },
    { emoji: '🛠️', label: 'Skills' },
    { emoji: '🧠', label: 'Thinking' },
    { emoji: '🖼️', label: 'Images' }
  ];

  const toggleTag = (tag) => {
    if (activeTags.includes(tag)) {
      setActiveTags(activeTags.filter(t => t !== tag));
    } else {
      setActiveTags([...activeTags, tag]);
    }
  };

  return (
    <div style={{ display: 'flex', flexWrap: 'wrap', gap: '8px' }}>
      {tags.map(tag => (
        <button
          key={tag.emoji}
          onClick={() => toggleTag(tag.emoji)}
          style={{
            background: activeTags.includes(tag.emoji) ? '#2563eb' : '#f3f4f6',
            color: activeTags.includes(tag.emoji) ? 'white' : 'black',
            border: '1px solid #ccc',
            borderRadius: '8px',
            padding: '6px 12px',
            cursor: 'pointer',
            fontSize: '0.9rem'
          }}
        >
          {tag.emoji} {tag.label}
        </button>
      ))}
    </div>
  );
}
