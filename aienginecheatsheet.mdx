import { useState } from "react";

<EngineFilter />

export const EngineFilter = () => {
  const engines = [
    { name: "Anthropic: Claude Sonnet 4.5", category: "Premium", tags: ["🧠","🎭","📚","🛠️"], description: "Favorite amongst roleplayers, 4.5 is intelligent, great for tool use, and Shape's default premium engine." },
    { name: "DeepSeek V3.2 Exp", category: "Premium", tags: ["🧠","📚","🛠️"], description: "The latest experimental model from DeepSeek." },
    { name: "XAI: Grok 4 Fast", category: "Premium", tags: ["🎭","📚"], description: "The faster, less expensive version of Grok 4." },
    { name: "OpenAI: GPT-5 Mini", category: "Free", tags: ["🎭","📚","🛠️"], description: "Balance of cost & capability." },
    { name: "OpenAI: GPT-4.1 Mini", category: "Free", tags: ["🎭","📚"], description: "Smaller, cheaper, faster version of GPT-4.1." },
    { name: "Grok 3 Mini", category: "Free", tags: ["🧠","🎭","📚","🛠️","👤"], description: "Amazing at following prompts, very fast, Swiss army knife of engines." },
  ];

  const allTags = [...new Set(engines.flatMap(e => e.tags))];
  const [selectedTag, setSelectedTag] = useState("");

  // Filtered engines based on selected tag
  const filtered = selectedTag === ""
    ? engines
    : engines.filter(engine => engine.tags.includes(selectedTag));

  // Separate by category
  const premiumEngines = filtered.filter(e => e.category === "Premium");
  const freeEngines = filtered.filter(e => e.category === "Free");

  return (
    <div>
  {/* Tag Dropdown */}
  <div style={{ marginBottom: "15px" }}>
    <select
      value={selectedTag}
      onChange={e => setSelectedTag(e.target.value)}
      style={{
        padding: "5px 12px",
        border: "1px solid #007bff",
        borderRadius: "5px",
        cursor: "pointer",
      }}
    >
      <option value="">All</option>
      {allTags.map(tag => (
        <option key={tag} value={tag}>{tag}</option>
      ))}
    </select>
  </div>

  {/* Premium Engines Section */}
  {filtered.some(e => e.category === "Premium") && (
    <div>
      <h2>Premium Engines</h2>
      <ul>
        {filtered.filter(e => e.category === "Premium").map(engine => (
          <li
            key={engine.name}
            style={{
              marginBottom: "10px",
              padding: "10px",
              borderRadius: "5px",
              transition: "background 0.2s",
              border: "1px solid #f59e0b"
            }}
          >
            <strong>{engine.name}</strong> [{engine.tags.join(", ")}]<br />
            {engine.description}
          </li>
        ))}
      </ul>
    </div>
  )}

  {/* Free Engines Section */}
  {filtered.some(e => e.category === "Free") && (
    <div>
      <h2>Free Engines</h2>
      <ul>
        {filtered.filter(e => e.category === "Free").map(engine => (
          <li
            key={engine.name}
            style={{
              marginBottom: "10px",
              padding: "10px",
              borderRadius: "5px",
              transition: "background 0.2s",
              border: "1px solid #10b981"
            }}
          >
            <strong>{engine.name}</strong> [{engine.tags.join(", ")}]<br />
            {engine.description}
          </li>
        ))}
      </ul>
    </div>
  )}
</div>

  );
};
