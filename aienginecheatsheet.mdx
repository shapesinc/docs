import { useState } from "react";

export const EngineFilter = () => {
  const engines = [
    { name: "Anthropic: Claude Sonnet 4.5", tags: ["🧠","🎭","📚","🛠️"], description: "Favorite amongst roleplayers, 4.5 is intelligent, great for tool use, and Shape's default premium engine." },
    { name: "OpenAI: GPT-5 Mini", tags: ["🎭","📚","🛠️"], description: "Balance of cost & capability." },
    { name: "Gemini 2.5 Pro (Latest)", tags: ["🎭","📚","👤","🛠️"], description: "Google’s state-of-the-art AI model." },
    { name: "OpenAI: GPT-4.1 Mini", tags: ["🎭","📚"], description: "Smaller, cheaper, faster version." },
    // Add the rest of your engines here
  ];

  // Determine Free vs Premium automatically
  const freeKeywords = ["Mini", "Lite", "Flash", "Turbo", "Small", "Nano"];
  const enginesWithCategory = engines.map(e => ({
    ...e,
    category: freeKeywords.some(k => e.name.includes(k)) ? "Free" : "Premium"
  }));

  const allTags = [...new Set(enginesWithCategory.flatMap(e => e.tags))];
  const [selectedTag, setSelectedTag] = useState("");

  const filtered = selectedTag === ""
    ? enginesWithCategory
    : enginesWithCategory.filter(engine => engine.tags.includes(selectedTag));

  const grouped = filtered.reduce((acc, engine) => {
    (acc[engine.category] = acc[engine.category] || []).push(engine);
    return acc;
  }, {});

  return (
    <div>
      <div style={{ marginBottom: "15px" }}>
        <select
          value={selectedTag}
          onChange={e => setSelectedTag(e.target.value)}
          style={{
            padding: "5px 12px",
            border: "1px solid #007bff",
            borderRadius: "5px",
            cursor: "pointer",
          }}
        >
          <option value="">All</option>
          {allTags.map(tag => (
            <option key={tag} value={tag}>{tag}</option>
          ))}
        </select>
      </div>

      {selectedTag === "" ? (
        <ul>
          {filtered.map(engine => (
            <li key={engine.name} style={{ marginBottom: "10px", padding: "10px", borderRadius: "5px" }}>
              <strong>{engine.name}</strong> [{engine.tags.join(", ")}]<br />
              {engine.description}
            </li>
          ))}
        </ul>
      ) : (
        <>
          {["Premium", "Free"].map(cat => (
            grouped[cat] && grouped[cat].length > 0 && (
              <div key={cat} style={{ marginBottom: "18px" }}>
                <h3>{cat}</h3>
                <ul>
                  {grouped[cat].map(engine => (
                    <li key={engine.name} style={{ marginBottom: "10px", padding: "10px", borderRadius: "5px" }}>
                      <strong>{engine.name}</strong> [{engine.tags.join(", ")}]<br />
                      {engine.description}
                    </li>
                  ))}
                </ul>
              </div>
            )
          ))}
        </>
      )}
    </div>
  );
};
