export const EngineFilter = () => {
  const engines = [
    { name: "Anthropic: Claude Sonnet 4.5", tags: ["🧠","🎭","📚","🛠️"], description: "Favorite amongst roleplayers, 4.5 is intelligent, great for tool use, and Shape's default premium engine." },
    { name: "DeepSeek V3.2 Exp", tags: ["🧠","📚","🛠️"], description: "The latest experimental model from DeepSeek." },
    { name: "XAI: Grok 4 Fast", tags: ["🎭","📚"], description: "The faster, less expensive version of Grok 4." },
    { name: "Gemini 2.5 Flash Image Preview", tags: ["🖼️"], description: "Image + text multimodal." },
    { name: "OpenAI: GPT-5 Chat", tags: ["🎭","👤"], description: "ChatGPT for Roleplayers and chatters." },
    { name: "OpenAI: GPT-5", tags: ["📚","🛠️"], description: "Top-tier reasoning, multimodal." },
    { name: "OpenAI: GPT-5 Mini", tags: ["🎭","📚","🛠️"], description: "Balance of cost & capability." },
    { name: "Anthropic: Claude Opus 4", tags: ["📚","🛠️"], description: "Long workflows, strong coding." },
    { name: "Anthropic: Claude Sonnet 4", tags: ["🎭","📚","🛠️"], description: "Improved coding & reasoning - community favorite for personality and roleplay style." },
    { name: "Anthropic: Claude Sonnet 3.7", tags: ["🎭","📚","🛠️"], description: "Strong instruction model, favorite amongst roleplays." },
    // ... rest of your engines
  ];

  const allTags = [...new Set(engines.flatMap(e => e.tags))];
  const [selectedTags, setSelectedTags] = useState([]);

  const toggleTag = (tag) => {
    setSelectedTags(prev =>
      prev.includes(tag)
        ? prev.filter(t => t !== tag)
        : [...prev, tag]
    );
  };

  const filtered = selectedTags.length === 0
    ? engines
    : engines.filter(engine =>
        selectedTags.every(tag => engine.tags.includes(tag))
      );

  return (
    <div>
      <div style={{ marginBottom: "15px" }}>
        <select
          multiple
          value={selectedTags}
          onChange={e => {
            const options = Array.from(e.target.selectedOptions).map(o => o.value);
            setSelectedTags(options);
          }}
          style={{
            padding: "5px 12px",
            border: "1px solid #007bff",
            borderRadius: "5px",
            cursor: "pointer",
            width: "200px",
            height: "100px",
          }}
        >
          {allTags.map(tag => (
            <option key={tag} value={tag}>{tag}</option>
          ))}
        </select>
      </div>

      <ul>
        {filtered.map(engine => (
          <li key={engine.name} style={{ marginBottom: "10px" }}>
            <strong>{engine.name}</strong> [{engine.tags.join(", ")}]<br />
            {engine.description}
          </li>
        ))}
      </ul>
    </div>
  );
};
