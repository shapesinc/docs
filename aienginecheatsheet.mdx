import { useState } from "react";

<EngineFilter />

export const EngineFilter = () => {
  const engines = [
    { name: "Anthropic: Claude Sonnet 4.5", tags: ["🧠","🎭","📚","🛠️"], description: "Favorite amongst roleplayers, 4.5 is intelligent, great for tool use, and Shape's default premium engine." },
    { name: "DeepSeek V3.2 Exp", tags: ["🧠","📚","🛠️"], description: "The latest experimental model from DeepSeek." },
    { name: "XAI: Grok 4 Fast", tags: ["🎭","📚"], description: "The faster, less expensive version of Grok 4." },
    // ... rest of your engines
  ];

  const allTags = [...new Set(engines.flatMap(e => e.tags))];
  const [selectedTags, setSelectedTags] = useState([]);

  const handleSelectChange = (event) => {
    const selectedOptions = Array.from(event.target.selectedOptions).map(opt => opt.value);
    setSelectedTags(selectedOptions);
  };

  const filtered = selectedTags.length === 0
    ? engines
    : engines.filter(engine =>
        selectedTags.every(tag => engine.tags.includes(tag))
      );

  return (
    <div>
      <div style={{ marginBottom: "15px" }}>
        <select multiple value={selectedTags} onChange={handleSelectChange} style={{ width: "200px", height: "100px" }}>
          {allTags.map(tag => (
            <option key={tag} value={tag}>
              {tag}
            </option>
          ))}
        </select>
      </div>

      <ul>
        {filtered.map(engine => (
          <li key={engine.name} style={{ marginBottom: "10px" }}>
            <strong>{engine.name}</strong> [{engine.tags.join(", ")}]<br />
            {engine.description}
          </li>
        ))}
      </ul>
    </div>
  );
};
