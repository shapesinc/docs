import { useState } from "react";

<EngineFilter />

export const EngineFilter = () => {
  const engines = [
    /* ... your engines array (unchanged) ... */
  ];

  const allTags = [...new Set(engines.flatMap(e => e.tags))];
  // Use an array for multi-select
  const [selectedTags, setSelectedTags] = useState([]);

  // If nothing selected, show all. Otherwise show engines that include ALL selected tags.
  const filtered = selectedTags.length === 0
    ? engines
    : engines.filter(engine => selectedTags.every(tag => engine.tags.includes(tag)));

  return (
    <div>
      <div style={{ marginBottom: "15px" }}>
        <select
          multiple
          value={selectedTags}
          onChange={e => {
            // Collect selected options; ignore the empty "All Tags" option if present
            const values = Array.from(e.target.selectedOptions)
              .map(o => o.value)
              .filter(v => v !== "");
            setSelectedTags(values);
          }}
          style={{
            padding: "5px 12px",
            border: "1px solid #007bff",
            borderRadius: "5px",
            cursor: "pointer",
            minHeight: "36px"
          }}
        >
          <option value="">All Tags</option>
          {allTags.map(tag => (
            <option key={tag} value={tag}>{tag}</option>
          ))}
        </select>
      </div>

      <ul>
        {filtered.map(engine => (
          <li key={engine.name} style={{ marginBottom: "10px" }}>
            <strong>{engine.name}</strong> [{engine.tags.join(", ")}]<br />
            {engine.description}
          </li>
        ))}
      </ul>
    </div>
  );
};
