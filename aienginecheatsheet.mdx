---

import { useState } from "react";

<EngineFilter />

export const EngineFilter = () => {
  const engines = [
    // Premium engines
    { name: "Anthropic: Claude Sonnet 4.5", tags: ["🧠","🎭","📚","🛠️"], description: "Favorite amongst roleplayers, 4.5 is intelligent, great for tool use, and Shape's default premium engine.", category: "Premium" },
    { name: "OpenAI: GPT-5", tags: ["📚","🛠️"], description: "Top-tier reasoning, multimodal.", category: "Premium" },
    { name: "Gemini 2.5 Pro (Latest)", tags: ["🎭","📚","👤","🛠️"], description: "Google’s state-of-the-art AI model for advanced reasoning, coding, mathematics, and scientific tasks, also popular for roleplayers.", category: "Premium" },
    { name: "Grok 4", tags: ["🎭","📚","👤","🛠️"], description: "Better coding & agent workflows.", category: "Premium" },
    
    // Free engines
    { name: "OpenAI: GPT-5 Mini", tags: ["🎭","📚","🛠️"], description: "Balance of cost & capability.", category: "Free" },
    { name: "OpenAI: GPT-4.1 Mini", tags: ["🎭","📚"], description: "Smaller, cheaper, faster version of GPT-4.1.", category: "Free" },
    { name: "Gemini 2.5 Flash (Lite)", tags: ["👤","🛠️"], description: "Fast, multilingual, fallback model.", category: "Free" },
    { name: "Grok 3 Mini", tags: ["🧠","🎭","📚","🛠️","👤"], description: "Amazing at following prompts, very fast, Swiss army knife of engines.", category: "Free" }
  ];

  const allTags = [...new Set(engines.flatMap(e => e.tags))];
  const [selectedTag, setSelectedTag] = useState("");

  const filtered = selectedTag === ""
    ? engines
    : engines.filter(engine => engine.tags.includes(selectedTag));

  // Group by category
  const grouped = filtered.reduce((acc, engine) => {
    if (!acc[engine.category]) acc[engine.category] = [];
    acc[engine.category].push(engine);
    return acc;
  }, {});

  return (
    <div>
      <div style={{ marginBottom: "15px" }}>
        <select
          value={selectedTag}
          onChange={e => setSelectedTag(e.target.value)}
          style={{
            padding: "5px 12px",
            border: "1px solid #007bff",
            borderRadius: "5px",
            cursor: "pointer",
          }}
        >
          <option value="">All</option>
          {allTags.map(tag => (
            <option key={tag} value={tag}>{tag}</option>
          ))}
        </select>
      </div>

      {Object.keys(grouped).map(category => (
        <div key={category} style={{ marginBottom: "20px" }}>
          <h3>{category}</h3>
          <ul>
            {grouped[category].map(engine => (
              <li
                key={engine.name}
                style={{
                  marginBottom: "10px",
                  padding: "10px",
                  borderRadius: "5px",
                  transition: "background 0.2s"
                }}
                onMouseLeave={e => e.currentTarget.style.background = "transparent"}
              >
                <strong>{engine.name}</strong> [{engine.tags.join(", ")}]<br />
                {engine.description}
              </li>
            ))}
          </ul>
        </div>
      ))}

    </div>
  );
};
