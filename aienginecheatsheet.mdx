import { useState } from "react";

<EngineFilter />

export const EngineFilter = () => {
  const engines = [
    // Premium Engines
    { name: "Anthropic: Claude Sonnet 4.5", tags: ["🧠","🎭","📚","🛠️"], description: "Favorite amongst roleplayers, 4.5 is intelligent, great for tool use, and Shape's default premium engine.", category: "Premium" },
    { name: "DeepSeek V3.2 Exp", tags: ["🧠","📚","🛠️"], description: "The latest experimental model from DeepSeek.", category: "Premium" },
    { name: "XAI: Grok 4 Fast", tags: ["🎭","📚"], description: "The faster, less expensive version of Grok 4.", category: "Premium" },
    { name: "Gemini 2.5 Flash Image Preview", tags: ["🖼️"], description: "Image + text multimodal.", category: "Premium" },
    { name: "OpenAI: GPT-5 Chat", tags: ["🎭","👤"], description: "ChatGPT for Roleplayers and chatters.", category: "Premium" },
    // ... include all other Premium engines here, category: "Premium"
    
    // Free Engines
    { name: "OpenAI: GPT-5 Nano", tags: ["🎭","📚"], description: "Low latency, huge context support.", category: "Free" },
    { name: "Grok 3 Mini", tags: ["🧠","🎭","📚","🛠️","👤"], description: "Extremely smart and fast, Swiss army knife of engines.", category: "Free" },
    { name: "Llama 4 Maverick", tags: ["🎭","📚","🛠️"], description: "17B expert mixture, multimodal.", category: "Free" },
    // ... include all other Free engines here, category: "Free"
  ];

  const allTags = [...new Set(engines.flatMap(e => e.tags))];
  const [selectedTags, setSelectedTags] = useState([]);

  const toggleTag = (tag) => {
    setSelectedTags(prev =>
      prev.includes(tag)
        ? prev.filter(t => t !== tag)
        : [...prev, tag]
    );
  };

  const filtered = selectedTags.length === 0
    ? engines
    : engines.filter(engine =>
        selectedTags.every(tag => engine.tags.includes(tag))
      );

  const categories = ["Premium", "Free"];

  return (
    <div>
      {/* Tag buttons */}
      <div style={{ marginBottom: "15px" }}>
        {allTags.map(tag => (
          <button
            key={tag}
            onClick={() => toggleTag(tag)}
            style={{
              margin: "5px",
              padding: "5px 12px",
              border: "1px solid #007bff",
              borderRadius: "5px",
              backgroundColor: selectedTags.includes(tag) ? "#007bff" : "#fff",
              color: selectedTags.includes(tag) ? "#fff" : "#007bff",
              cursor: "pointer",
            }}
          >
            {tag}
          </button>
        ))}
      </div>

      {/* Tables by category */}
      {categories.map(category => {
        const enginesByCategory = filtered.filter(e => e.category === category);
        return (
          <div key={category} style={{ marginBottom: "30px" }}>
            <h2>{category} Engines</h2>
            <table style={{ width: "100%", borderCollapse: "collapse" }}>
              <thead>
                <tr>
                  <th style={{ textAlign: "left", borderBottom: "2px solid #ddd", padding: "8px" }}>Engine</th>
                  <th style={{ textAlign: "left", borderBottom: "2px solid #ddd", padding: "8px" }}>Tags</th>
                  <th style={{ textAlign: "left", borderBottom: "2px solid #ddd", padding: "8px" }}>Description</th>
                </tr>
              </thead>
              <tbody>
                {enginesByCategory.map(engine => (
                  <tr key={engine.name}>
                    <td style={{ padding: "8px", borderBottom: "1px solid #eee" }}>{engine.name}</td>
                    <td style={{ padding: "8px", borderBottom: "1px solid #eee" }}>{engine.tags.join(" ")}</td>
                    <td style={{ padding: "8px", borderBottom: "1px solid #eee" }}>{engine.description}</td>
                  </tr>
                ))}
              </tbody>
            </table>
          </div>
        );
      })}
    </div>
  );
};
