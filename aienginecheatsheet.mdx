];

export default function FilterableEngines() {
  const [activeTags, setActiveTags] = useState(new Set());

  function toggleTag(tag) {
    const newSet = new Set(activeTags);
    if (newSet.has(tag)) newSet.delete(tag);
    else newSet.add(tag);
    setActiveTags(newSet);
  }

  return (
    <div>
      <h2>Filter Engines</h2>
      <div style={{ display: "flex", gap: "8px", marginBottom: "1em" }}>
        {["🎭", "👤", "📚", "🛠️", "🧠", "🖼️"].map(tag => (
          <button
            key={tag}
            onClick={() => toggleTag(tag)}
            style={{
              padding: "6px 12px",
              borderRadius: "8px",
              border: "1px solid #ccc",
              cursor: "pointer",
              backgroundColor: activeTags.has(tag) ? "#2563EB" : "#f3f4f6",
              color: activeTags.has(tag) ? "white" : "black",
            }}
          >
            {tag}
          </button>
        ))}
      </div>

      <table style={{ width: "100%", borderCollapse: "collapse" }}>
        <thead>
          <tr>
            <th>Engine</th>
            <th>Tags</th>
            <th>Description</th>
          </tr>
        </thead>
        <tbody>
          {engines
            .filter(engine => [...activeTags].every(tag => engine.tags.includes(tag)))
            .map(engine => (
              <tr key={engine.name}>
                <td>{engine.name}</td>
                <td>{engine.tags.join(" ")}</td>
                <td>{engine.description}</td>
              </tr>
            ))}
        </tbody>
      </table>
    </div>
  );
}
