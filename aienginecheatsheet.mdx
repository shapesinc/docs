export function FilterButtons() {
  const [active, setActive] = useState([]);

  const tags = ["🎭", "👤", "📚", "🛠️", "🧠", "🖼️"];

  function toggle(tag) {
    setActive(prev =>
      prev.includes(tag) ? prev.filter(t => t !== tag) : [...prev, tag]
    );
  }

  // Update rows after every state change
  if (typeof window !== "undefined") {
    const rows = document.querySelectorAll("tbody tr");
    rows.forEach(row => {
      const tagsText = row.children[1]?.innerText || "";
      const match = active.every(t => tagsText.includes(t));
      row.classList.toggle("hidden", active.length && !match);
    });
  }

  return (
    <>
      {tags.map(tag => (
        <button
          key={tag}
          className={active.includes(tag) ? "active" : ""}
          onClick={() => toggle(tag)}
        >
          {tag}
        </button>
      ))}
    </>
  );
}
