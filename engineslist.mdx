  // State
  let activeFilters = [];

  // Render filter buttons
  function renderButtons() {
    const buttonsContainer = document.getElementById('capability-buttons');
    let html = '';
    capabilities.forEach(cap => {
      html += `<button style="padding: 8px 16px; border-radius: 8px; font-size: 14px; font-weight: 500; background: ${activeFilters.includes(cap.key) ?     });
    buttonsContainer.innerHTML = html;
  }

  // Toggle filter
  window.toggleFilter = function(key) {
    activeFilters = activeFilters.includes(key) ? activeFilters.filter(f => f !== key) : [...activeFilters, key];
    updateResults();
  };

  // Filter engines
  function filterEngines(engines) {
    if (activeFilters.length === 0) return engines;
    return engines.filter(engine => activeFilters.every(filter => engine.capabilities[filter]));
  }

  // Render results
  function updateResults() {
    const filteredPremium = showPremium ? filterEngines(enginesData.premium) : [];
    const filteredFree = showFree ? filterEngines(enginesData.free) : [];
    const totalResults = filteredPremium.length + filteredFree.length;

    const clearSection = document.getElementById('clear-filters');
    clearSection.style.display = activeFilters.length > 0 ? 'block' : 'none';
    document.getElementById('result-count').innerText = `(${totalResults} ${totalResults === 1 ? 'result' : 'results'})`;

    const resultsContainer = document.getElementById('results');
    let html = '';

    if (showPremium && filteredPremium.length > 0) {
      html += `
        <h2 style="font-size: 24px; font-weight: 700; margin-bottom: 16px; color: #111827;">Premium (${filteredPremium.length})</h2>
        <div style="display: grid; gap: 16px; margin-bottom: 24px; ${window.innerWidth >= 768 ? 'grid-template-columns: repeat(2, 1fr);' : ''}">
      `;
      filteredPremium.forEach(engine => {
        html += `
          <div style="padding: 20px; border: 1px solid #e5e7eb; border-radius: 8px; background: #fff;">
            <div style="display: flex; align-items: center; margin-bottom: 8px;">
              <h4 style="font-size: 18px; font-weight: 600; color: #111827;">${engine.name}</h4>
              <span style="font-size: 12px; background: #fef9c3; color: #854d0e; padding: 4px 8px; border-radius: 9999px; margin-left: 8px;">Premium</span>
            </div>
            <p style="font-size: 14px; color: #6b7280; margin-bottom: 12px;">${engine.description || 'No description available.'}</p>
            <div style="display: flex; flex-wrap: wrap; gap: 8px;">
              ${Object.keys(engine.capabilities).map(key => engine.capabilities[key] ? `<span style="font-size: 12px; background: #eff6ff; color: #1D4ED8; padding: 4px 8px; border-radius: 9999px;">${capabilities.find(c => c.key === key).icon} ${capabilities.find(c => c.key === key).label}</span>` : '').join('')}
            </div>
          </div>
        `;
      });
      html += '</div>';
    }

    if (showFree && filteredFree.length > 0) {
      html += `
        <h2 style="font-size: 24px; font-weight: 700; margin-bottom: 16px; color: #111827;">Free (${filteredFree.length})</h2>
        <div style="display: grid; gap: 16px; ${window.innerWidth >= 768 ? 'grid-template-columns: repeat(2, 1fr);' : ''}">
      `;
      filteredFree.forEach(engine => {
        html += `
          <div style="padding: 20px; border: 1px solid #e5e7eb; border-radius: 8px; background: #fff;">
            <div style="display: flex; align-items: center; margin-bottom: 8px;">
              <h4 style="font-size: 18px; font-weight: 600; color: #111827;">${engine.name}</h4>
            </div>
            <p style="font-size: 14px; color: #6b7280; margin-bottom: 12px;">${engine.description || 'No description available.'}</p>

            </div>
          </div>
        `;
      });
      html += '</div>';
    }

    if (totalResults === 0) {
      html += `
        <div style="text-align: center; padding: 48px 0;">
          <p style="font-size: 18px; color: #6b7280;">No engines match your selected filters.</p>
          <button style="font-size: 14px; color: #3B82F6; text-decoration: underline; cursor: pointer; background: none; border: none; margin-top: 12px;" onclick="clearFilters()">Clear filters to see all engines</button>
        </div>
      `;
    }

    resultsContainer.innerHTML = html;
    renderButtons();
  }

  // Clear filters
  window.clearFilters = function() {
    activeFilters = [];
    updateResults();
  };

  // Initialize
  document.getElementById('show-premium').addEventListener('change', function(e) {
    showPremium = e.target.checked;
    updateResults();
  });
  document.getElementById('show-free').addEventListener('change', function(e) {
    showFree = e.target.checked;
    updateResults();

    activeFilters = [];
    updateResults();

## Key
- 🧠 **Thinking**: Supports advanced reasoning or slower, deeper processing.
- 🎭 **Roleplay**: Great for creative, character-driven interactions.
- 📚 **Intelligence**: High knowledge retrieval and factual accuracy.
- 🛠️ **Skills**: Enabled for tool use or specialized tasks (e.g., coding, math).
- 👤 **Human-Like**: Conversational, friendly, or mimics human behavior.
- 🖼️ **Images**: Supports image processing or multimodal tasks.
